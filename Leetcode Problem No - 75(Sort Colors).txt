class Solution {
public:
    void sortColors(vector<int>& color) {
        
         int n = color.size();

  vector<int>frequency(3 , 0);

  for(int i = 0; i < n; i++){ 
  
     frequency[color[i]]++;
}

color.clear();

for(int i = 0; i < 3; i++)
{
    while(frequency[i]--){ 
   
  color.push_back(i);
}
}

}
};
