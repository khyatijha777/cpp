class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int prod = 1;
        int len = nums.size();
        vector<int> output(len, 1);
        for (int i = 0; i < len; ++i) {
            output[i] = prod;
            prod *= nums[i];
        }
        prod = 1;
        for (int i = len - 1; i >= 0; --i) {
            output[i] *= prod;
            prod *= nums[i];
        }
        return output;
    }
};
